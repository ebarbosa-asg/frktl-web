# Brief: T07 — Replace IntersectionObserver with Motion Scroll Animations + Spring Physics

**Wave:** 3 | **Complexity:** 2/10 | **Priority:** Must | **Depends on:** T02, T03

---

## Context

The source `frktl-explainer.jsx` uses a manual `IntersectionObserver` + CSS class toggle for all section entrance animations. This task replaces that pattern with Motion (`motion/react` v11+) `useInView` and `motion.div` wrappers, upgrading entrance animations from linear `ease` to spring physics.

The Motion package was already installed in T01 (`npm install motion`). This task wires it up.

**What changes:** The animation mechanism (IntersectionObserver → Motion). The visual output is the same: sections fade in and slide up when scrolled into view.

**What does NOT change:** The SVG SMIL animations. The 10 SVG components still receive an `animate: boolean` prop — this task provides that boolean via `useInView` instead of via the old `isVisible[idx]` state. The SMIL animations themselves (in the SVG files from T03) are untouched.

**The roadmap phase cycling:** The `activePhase` state that cycles 0→4 every 3 seconds (driving `RoadmapTimelineSVG`) must be preserved. This lives in the `RoadmapSection` component and uses a plain `setInterval` in `useEffect` — it is not replaced by Motion.

## Source Pattern Being Replaced

### Original pattern (frktl-explainer.jsx, lines 441–455):
```jsx
// Old: manual IntersectionObserver
const [isVisible, setIsVisible] = useState({});
useEffect(() => {
  const observer = new IntersectionObserver(
    (entries) => {
      entries.forEach((entry) => {
        const idx = Number(entry.target.dataset.idx);
        if (entry.isIntersecting) {
          setIsVisible((prev) => ({ ...prev, [idx]: true }));
          setActiveSection(idx);
        }
      });
    },
    { threshold: 0.15 }
  );
  sectionRefs.current.forEach((el) => el && observer.observe(el));
  return () => observer.disconnect();
}, []);
```

### Original CSS animation (injected style tag):
```css
.section-enter { opacity: 0; transform: translateY(50px); transition: opacity 0.8s ease, transform 0.8s ease; }
.section-visible { opacity: 1; transform: translateY(0); }
```

### Target pattern (Motion):
```tsx
import { motion, useInView } from "motion/react";

function AnimatedSection({ children }: { children: React.ReactNode }) {
  const ref = useRef<HTMLDivElement>(null);
  const isInView = useInView(ref, { once: true, amount: 0.15 });
  return (
    <motion.div
      ref={ref}
      initial={{ opacity: 0, y: 60 }}
      animate={isInView ? { opacity: 1, y: 0 } : { opacity: 0, y: 60 }}
      transition={{ type: "spring", stiffness: 100, damping: 20 }}
    >
      {children}
    </motion.div>
  );
}
```

## Technical Specification

### File: components/sections/SectionWrapper.tsx

This is the reusable wrapper that every content section uses. It provides the Motion animation context.

```tsx
"use client";

import { useRef } from "react";
import { motion, useInView } from "motion/react";

interface SectionWrapperProps {
  children: (isInView: boolean) => React.ReactNode;
  className?: string;
  id?: string;
}

// Variants for staggered children (used by sections with multiple cards/items)
export const staggerContainer = {
  hidden: { opacity: 0 },
  visible: {
    opacity: 1,
    transition: {
      staggerChildren: 0.08,
    },
  },
};

export const fadeUpItem = {
  hidden: { opacity: 0, y: 30 },
  visible: {
    opacity: 1,
    y: 0,
    transition: {
      type: "spring" as const,
      stiffness: 100,
      damping: 20,
    },
  },
};

export function SectionWrapper({ children, className, id }: SectionWrapperProps) {
  const ref = useRef<HTMLElement>(null);
  const isInView = useInView(ref, { once: true, amount: 0.15 });

  return (
    <motion.section
      ref={ref}
      id={id}
      className={className}
      initial={{ opacity: 0, y: 60 }}
      animate={isInView ? { opacity: 1, y: 0 } : { opacity: 0, y: 60 }}
      transition={{ type: "spring", stiffness: 100, damping: 20 }}
    >
      {/* Children receive the isInView boolean to pass to SVG components as animate prop */}
      {children(isInView)}
    </motion.section>
  );
}
```

**Render children as a function pattern** (`children: (isInView: boolean) => React.ReactNode`): This allows the section content to receive the `isInView` boolean and pass it to SVG illustration components as `animate={isInView}`. This is the direct replacement for the old `animate={vis}` prop.

### File: components/sections/HeroSection.tsx

The hero section gets a special treatment: scroll-linked parallax on the background glow using `useScroll` + `useTransform`.

```tsx
"use client";

import { useRef } from "react";
import { motion, useScroll, useTransform } from "motion/react";

export function HeroSection() {
  const ref = useRef<HTMLDivElement>(null);
  const { scrollYProgress } = useScroll({
    target: ref,
    offset: ["start start", "end start"],
  });

  // Background glow moves at 0.3× scroll rate (subtle parallax)
  const glowY = useTransform(scrollYProgress, [0, 1], ["0%", "30%"]);

  return (
    <div
      ref={ref}
      className="min-h-[100dvh] flex flex-col items-center justify-center text-center px-6 py-10 relative overflow-hidden"
    >
      {/* Parallax background glow */}
      <motion.div
        style={{ y: glowY }}
        className="absolute w-[600px] h-[600px] rounded-full pointer-events-none"
        aria-hidden="true"
        css={{
          background: "radial-gradient(circle, var(--color-teal) 0%, transparent 70%)",
          opacity: 0.08,
          top: "50%",
          left: "50%",
          transform: "translate(-50%, -50%)",
        }}
      />

      {/* Hero content — static (no entrance animation; it's above the fold) */}
      {/* Spline 3D scene goes here (T09 will add HeroScene component) */}
      {/* T08 will style this section with the 50s editorial treatment */}
    </div>
  );
}
```

### File: hooks/useScrollParallax.ts

```ts
"use client";

import { useScroll, useTransform, MotionValue } from "motion/react";
import { useRef, RefObject } from "react";

interface UseScrollParallaxOptions {
  speed?: number; // 0.3 = moves at 30% of scroll speed
  direction?: "y" | "x";
}

export function useScrollParallax(
  options: UseScrollParallaxOptions = {}
): {
  ref: RefObject<HTMLElement | null>;
  value: MotionValue<string>;
} {
  const { speed = 0.3, direction = "y" } = options;
  const ref = useRef<HTMLElement>(null);
  const { scrollYProgress } = useScroll({
    target: ref,
    offset: ["start start", "end start"],
  });

  const range = speed * 100;
  const value = useTransform(
    scrollYProgress,
    [0, 1],
    direction === "y" ? [`0%`, `${range}%`] : [`0%`, `${range}%`]
  );

  return { ref, value };
}
```

### RoadmapSection — preserve activePhase cycling

The `activePhase` cycling must be preserved in whatever component renders the roadmap section. Pattern:

```tsx
"use client";

import { useState, useEffect } from "react";
import { RoadmapTimelineSVG } from "@/components/illustrations";

export function RoadmapSection({ animate }: { animate: boolean }) {
  const [activePhase, setActivePhase] = useState(0);

  // Preserve the 3-second cycling behavior from the original
  useEffect(() => {
    const timer = setInterval(() => {
      setActivePhase((p) => (p + 1) % 5);
    }, 3000);
    return () => clearInterval(timer);
  }, []);

  return (
    <div>
      {/* ... section content ... */}
      <RoadmapTimelineSVG animate={animate} activePhase={activePhase} />
    </div>
  );
}
```
